def attr_to_hash(nokogiri_attrs)
  attributes = {}
  nokogiri_attrs.each {
    |k,v|
    attributes[k] = v.to_s
  }
  if attributes['lang']
    attributes['xml:lang'] = attributes['lang'];
    attributes.delete('lang')
  end
  attributes
end

def ns_prefix(ns)
  prefix = raw_prefix(ns)
  prefix.nil? ? nil : prefix.sub(/^xmlns:/,'')
end
