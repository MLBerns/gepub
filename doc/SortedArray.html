def initialize(*args, &sort_by)
  @sort_by = sort_by || Proc.new { |x,y| x <=> y }
  super(*args)
  self.sort!() &sort_by
end

def insert(i, v)
  insert_before = index(find { |x| @sort_by.call(x, v) == 1 })
  super(insert_before ? insert_before : -1, v)
end
